    public static void main(String[] args) {

        int n = 10;

            int min_index = i; // κ°€μ¥ μ‘μ€ μ›μ†μ μΈλ±μ¤

            int temp = arr[i];

        }

            System.out.print(arr[i] + " ");
        }
    }

}
